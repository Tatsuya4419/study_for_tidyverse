Study for Tidyverse

library(tidyverse)
df <- tibble(
  A = c(1, 2, 3),
  B = c(4 ,5, 6),
  C = c(7, 8, 9)
  )
df
## # A tibble: 3 x 3
##       A     B     C
##   <dbl> <dbl> <dbl>
## 1     1     4     7
## 2     2     5     8
## 3     3     6     9

map系関数の中で列名を参照する

df %>%
  map2_df(., names(.), ~paste0(.x, .y))
## # A tibble: 3 x 3
##   A     B     C    
##   <chr> <chr> <chr>
## 1 1A    4B    7C   
## 2 2A    5B    8C   
## 3 3A    6B    9C

行ごとに演算

このようにすると全要素を計算に使ってしまう。

df %>% 
  rowwise() %>% 
  mutate(sum = sum(.))
## # A tibble: 3 x 4
## # Rowwise: 
##       A     B     C   sum
##   <dbl> <dbl> <dbl> <dbl>
## 1     1     4     7    45
## 2     2     5     8    45
## 3     3     6     9    45

across()関数を使うと行ごとの処理になる。

df %>% 
  rowwise() %>% 
  mutate(sum = sum(across()))
## # A tibble: 3 x 4
## # Rowwise: 
##       A     B     C   sum
##   <dbl> <dbl> <dbl> <dbl>
## 1     1     4     7    12
## 2     2     5     8    15
## 3     3     6     9    18

pmap()でも同じことができる。

df %>% 
  mutate(sum = pmap_dbl(., sum))
## # A tibble: 3 x 4
##       A     B     C   sum
##   <dbl> <dbl> <dbl> <dbl>
## 1     1     4     7    12
## 2     2     5     8    15
## 3     3     6     9    18

特定の列に対して関数を適用

formulaを使うことでシンプルに記述できる。

df %>%
  mutate_at(vars(-A), ~ . + A)
## # A tibble: 3 x 3
##       A     B     C
##   <dbl> <dbl> <dbl>
## 1     1     5     8
## 2     2     7    10
## 3     3     9    12
df %>% 
  mutate(
    across(-A, ~ . + A)
  )
## # A tibble: 3 x 3
##       A     B     C
##   <dbl> <dbl> <dbl>
## 1     1     5     8
## 2     2     7    10
## 3     3     9    12

NSEによる変数指定に環境変数を使いたいときの方法いろいろ。

cols <- c("A", "B")

df %>% 
  group_by(across(all_of(cols)))
## # A tibble: 3 x 3
## # Groups:   A, B [3]
##       A     B     C
##   <dbl> <dbl> <dbl>
## 1     1     4     7
## 2     2     5     8
## 3     3     6     9
col <- "A"

df %>% 
  filter(.data[[col]] > 1)
## # A tibble: 2 x 3
##       A     B     C
##   <dbl> <dbl> <dbl>
## 1     2     5     8
## 2     3     6     9
col <- "D"

df %>% 
  mutate({{col}} := C + 3)
## # A tibble: 3 x 4
##       A     B     C     D
##   <dbl> <dbl> <dbl> <dbl>
## 1     1     4     7    10
## 2     2     5     8    11
## 3     3     6     9    12
